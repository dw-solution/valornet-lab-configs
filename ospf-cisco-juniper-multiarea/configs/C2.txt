! ===================================================================
! C2 - Cisco - Area 0 Backbone
! Includes e0/2 stitching link to J2 (fixes discontiguous Area 0)
! ===================================================================

enable
configure terminal

hostname C2

interface Loopback0
 ip address 2.2.2.2 255.255.255.255

interface Ethernet0/0
 description To C1 (Area 0)
 ip address 192.168.1.2 255.255.255.252
 no shutdown

interface Ethernet0/1
 description To C3 (Area 0)
 ip address 192.168.1.5 255.255.255.252
 no shutdown

interface Ethernet0/2
 description To J2 (Area 0 - backbone stitching link)
 ip address 192.168.1.25 255.255.255.252
 no shutdown

router ospf 1
 router-id 2.2.2.2
 network 2.2.2.2 0.0.0.0 area 0
 network 192.168.1.0 0.0.0.3 area 0
 network 192.168.1.4 0.0.0.3 area 0
 network 192.168.1.24 0.0.0.3 area 0

end
write memory
